常用工具包

import cv2

图像操作

图片读取

img = cv2.imread('xxx.jpg') # 读取图像 数据类型
img = cv2.imread('xxx.jpg'， cv2.IMREAD_COLOR) # 读取彩色图像
img = cv2.imread('xxx.jpg'， cv2.IMREAD_GRAYSCALE) # 读取灰度图像
cv2.imshow('xxx', img) # 显示图像（窗口名，图像）
cv2.waitKey(0)
cv2.destoryAllWindows()
cv2.imwrite('xxx', img) # 保存图像

# 查看图片的属性
type(img)
img.shape
img.size
img.dtype
